\section{Isomorphism Theorems}

\subsection*{Subgroups}

\begin{definition}
    A nonemptyset \(S\subset G\) is a \emph{subgroup} of  \(G\) if (i) \(s \in S\) implies \(s^{-1} \in S\) and (ii) \(s,t\in G\) imply \(st\in G\). We denote \(S\leq G\). 
\end{definition}

\begin{theorem}
    \textbf{2.1.} \((S,*)\leq (G,*)\) is a group in its own right. \comentario{\(*|_{S \times S}\) + definition}
\end{theorem}

\begin{theorem}
    \textbf{2.2.} \(S\leq G, \ \leftrightharpoons \ \id \in S \ \text{and } s,t \in S\) imply \(st^{-1} \in S\).  
\end{theorem}

\begin{definition}
    Given \(a \in G, \ \{a^m : m \in \Z\}=: \langle a \rangle \leq G\) is the \emph{cyclic subgroup generated by} \(a\); \(G\) is \emph{cyclic} if \(\exists a : G = \langle a \rangle\) and \(|\langle a \rangle|\) is the \emph{order} of \(a\). 
\end{definition}

\begin{theorem}
    \textbf{2.3.} If \(|\langle a \rangle | = m < \infty\), then \(m = \min \{k : a^k = \id, \ k \in \N\}\). 
\end{theorem}

\demo{
    It's clear that \(\exists k \in \N : a^k = \id\). Let \(\{a^j\}_{j=0}^{k-1} : \forall i < l \leq k-1, \ a^i \neq a^l\), prove that \(a^k = \id\) (by contradiction). Finally prove \(\langle a \rangle = \{a^j\}\) using the division algorithm.  \hspace{-0.3cm}
}

\begin{corollary}
    \textbf{2.4.} If \(|G| < \infty\), then \(S \leq G \ \leftrightharpoons \ s,t \in S\) imply \(st \in S\). \textcolor{gray}{(\(\Leftarrow\)) If \(s \in S\), then \(\langle s\rangle \leq S\), \(\exists m \in \N: s^m = \id\) and \(s^{-1} = s^{m-1} \in S\) }  
\end{corollary}

\begin{note}
    Clearly \(\{\id\}< H <G \leq G\), in this case \(H\) is a \emph{proper subgroup}. 
\end{note}

\ejemplo{
    \centering Kernel and Image of \(f: G \to H\) homomorphism 
    \tcblower 
    \[\text{\scriptsize $\bullet $ \ \ }\{a \in G : f(a) = \id_{_H}\} =: \ker f\leq G;\hspace{5cm}\] 
    \[\hspace{4cm}\text{\scriptsize $\bullet $ \ \ }\{h \in H : h = f(a), \ a \in G\} =:\operatorname{Im}f\leq H.\]  
}

\begin{theorem}
    \textbf{2.5.} Let \((S_i)_{i\in I} : S_i \leq G\), then \(\bigcap S_i \leq G\). \comentario{Mogollas }
\end{theorem}

\begin{definition}
    Given \(X \subset G\), \(\langle X \rangle:= \bigcap \{S : S \leq G \text{ and } X \subset S\} \) is the subgroup of \(G\) generated by \(X\). If \(|X|= n < \infty\), we write \(\langle X \rangle = \langle a_1, a_2, \ldots, a_n\rangle\).  
\end{definition}

\begin{note}
    If \(H, K \leq G\), then we denote \(\langle H \cup K \rangle \) as \(H \lor K\).
\end{note}

\begin{definition}
    If \(\varnothing \neq X\subset G\), then a \emph{word} on \(X\) is an expression of the form \(x_1^{e_1} x_2^{e_2}\ldots x_n^{e_n}= w \in G\), where \(x_j \in X\) and \(e_j = \pm 1 \).    
\end{definition}

\begin{theorem}
    \textbf{2.7.} \(\langle X \rangle \leq G\) is precisely the set of all words on \(X\). \comentario{Let \(W\) the set of all words, prove that \(W \leq G\) is the smallest \(: X \subset W\)}
\end{theorem}

\subsection*{Lagrange's Theorem}

\begin{definition}
    Let \(S\leq G\) and \(t \in G\). The \emph{right coset} of \(S\) in \(G\) is the subset \(St := \{st : s \in S\}\), we call \(t \) a \emph{representative} of \(St\) (similarly we define \(tS\)). 
\end{definition}

\ejemplo{
\begin{example}
    \(\langle n \rangle =: S < \Z\). Given \(a \in \Z, \ a + S = \{a + qn : q \in \Z\} = [a] \in \Z_n \).   
\end{example}
}

\begin{note}
    In general \(tS\) and \(St\) need not coincide, may be different. %, let \(H := \langle (1 \ \ 2) \rangle \leq S_3\). 
\end{note}

\hypertarget{lemma28}{}
\begin{lemma}
    \textbf{2.8.} \(Sa = Sb \ (\text{or }aS = bS) \ \leftrightharpoons \ ab^{-1} \in S \ (\text{or }b^{-1}a \in S)\). 
\end{lemma}

\demo{(\(\Rightarrow \)) \(\id a \in Sa = Sb\), so \(\exists s \in S : a = sb \ \leftrightharpoons \ ab^{-1} = s \in S\). (\(\Leftarrow\)) Prove two inclusions using \(Sa \ni x = sa =  s\sigma b \in Sb: \sigma = a b^{-1}\) (similarly for the other).}

\hypertarget{thm29}{}
\begin{theorem}
    \textbf{2.9.} For any two right (or two left) cosets of \(S\) in \(G\) are either identical or disjoint. \comentario{If \(x = sa = tb\), then \(ab^{-1} = s^{-1}t \in S\), result follows from \(\blacksquare\) \hyperlink{lemma28}{2.8.} }
\end{theorem}

\begin{theorem}
    \textbf{2.10.} The number of right cosets is equal to the number of left cosets of \(S\) in \(G\). 
\end{theorem}

\demo{
    Proof that \(f: \mathscr{L} \to \mathscr{R}\) such that \(Sa \mapsto a^{-1} S\) is a well defined bijection. 
}

\begin{definition}
    The \emph{index} of \(S\) in \(G\), denoted by \([G:S]\), is the number of right cosets. 
\end{definition}

\hypertarget{thm211}{}
\begin{theorem}
    \textbf{2.11. (\emph{Lagrange})} If \(|G| < \infty\) and \(S \leq G\), then \(|S| \) divides \( |G|\) and \([G:S] = \nicefrac{|G|}{|S|}\).  
\end{theorem}

\demo{
    By \(\blacksquare\) \hyperlink{thm29}{2.9.} \(G = St_1 \sqcup St_2 \sqcup \ldots \sqcup St_n\), and so \(|G| = \sum |St_j| \), how \(f_j : S \mapsto St_j\) is bijection, \(|St_j| = |S|\) and \(|G| = n |S|\), where \(n = [G:S]\).   
}

\begin{corollary}
    \textbf{2.12.} If \(|G|< \infty\) and \(a \in G\), then the order of \(a\) divides \(|G|\). Moreover, if \(|G| = p \) prime, then \(|G|\) is a cyclic group. \comentario{Takes \(a \neq \id \), then \(|\langle a \rangle|\) divides \(p \ \leftrightharpoons \ |\langle a \rangle | = p\)}
\end{corollary}

\renewcommand{\mod}[1]{\ (\operatorname{mod} #1)}
\begin{corollary}
    \textbf{2.14. (\emph{Fermat})} If \(p\) is prime and \(a \in \Z\), then \(a^p \equiv a \mod{p}\). 
\end{corollary}

\demo{Let \(\F = \Z_p\) and \(G = (\F^\times, \cdot )\). If \([a] = [0]\), then clearly \([a]^p = [a] = [0] \). Otherwise, \([a]^{p-1} = [\id]\), so \([a]^{p} = [a]\), any case \(a^{p} \equiv a \mod{p}\).}

\subsection*{Cyclic Groups}

\hypertarget{lemma215}{}
\begin{lemma}
    \textbf{2.15.} If \(G = \langle a\rangle : |G|= n < \infty\), then \(\exists ! S\leq G : |S| = d \) divides \(n\). 
\end{lemma}

\demo{Let \(H = \langle b\rangle : |H|=d\) \ (\textcolor{rojoscuro}{Ex. 18}), then \(b^d = \id = {(a^{m})}^d\), so \(\exists k \in \N : md = nk\) \ (\textcolor{rojoscuro}{Ex. 13}). Finally, \(b = a^m = (a^{\nicefrac{n}{d}})^k\)  and \(H \leq \langle a^{\nicefrac{n}{d}}\rangle =: S \leq G\) (\textcolor{rojoscuro}{Ex. 11}). Since \( |S| = d \), we have \(H= S\). }

\begin{definition}
    \centering \(\displaystyle \varphi(n) := \# \{k\in \N : k < n \text{ and } (k,n) = 1\}\).
\end{definition}
\hypertarget{thm216}{}
\begin{theorem}
    \textbf{2.16.} If \(n\) is a positive integer, then \(\displaystyle n = \sum_{d \hspace{0.1cm} \mid \hspace{0.1cm}  n} \varphi(d)\).   
\end{theorem}

\demo{Let \(G = \langle a\rangle : |G|=n\), then \(G = \bigsqcup \operatorname{gen}C : C \leq G\). By {\scriptsize\(\square\)} \hyperlink{lemma215}{2.15.} \(\forall d \in \operatorname{Div}(n), \ \exists ! C_d \leq G\), so \(|G| = \underset{d \ \mid \ n}{\sum} |\operatorname{gen}C_d| =\underset{d \ \mid \ n}{\sum}\varphi(d)\) (\textcolor{rojoscuro}{Ex. 20}). \vspace{-0.1cm} }

\begin{theorem}
    \textbf{2.17.} \(G : |G| = n\) is cyclic \(\leftrightharpoons \ \forall d : d  \mid  n, \ \exists ! C_d \leq G : |C_d| = d\).  
\end{theorem}

\demo{(\(\Rightarrow\)) {\scriptsize \(\square\) } \hyperlink{lemma215}{2.15.}. (\(\Leftarrow\)) By \(\blacksquare\) \hyperlink{thm216}{2.16.} \(n = \sum |\operatorname{gen}C| \leq \sum \varphi(d) = n\), so \(G\) must have a subgroup of order \(n \ \leftrightharpoons \ G\) is cyclic. }

\begin{theorem}
    \textbf{2.18.} \(G\leq  \F^\times : |G| < \infty\) is cyclic. In  particular, \(\F^\times\) is cyclic if \(|\F| < \infty \). \comentario{The polynomial \(x^d - \id \) has almost \(d\) roots in \(\F[x]\), result follows from \(\blacksquare\) \hyperlink{thm217}{2.17.} }
\end{theorem}

\begin{theorem}
    \textbf{2.19.} Let \(p\) prime. A group \(G : |G| = p^n\) is cyclic \ \(\leftrightharpoons\) \ it's an abelian group and \(\exists ! S\leq G : |S| = p\).  \comentario{Look \cite[pag. 29]{stein}, needs (\textcolor{rojoscuro}{Ex. 13}) }
\end{theorem}

\subsection*{Normal Subgroups}

\begin{definition}
    \centering \(ST := \{st : s \in S, \ t \in T\} \subset G\).   
\end{definition}

\begin{theorem}
    \textbf{2.20. (\emph{Product Formula})} If \(S, T \leq G : |G|< \infty \), then \(|ST| |S \cap T| = |S||T|\). \footnote{\(ST\) need not to be a subgroup.} \comentario{Doesn't support sumaries, look if you want \cite[pag. 30]{stein}}
\end{theorem}

\begin{definition}
    \(K\leq G\) is a \emph{normal subgroup}, denoted by \(K\lhd G\), if \ \(\forall g \in G, \ gKg^{-1} = K\).  
\end{definition}

\begin{example}
    If \(f : G \to H\) is homomorphism, then \(\ker f \lhd G\). 
\end{example}

\begin{note}
    \(K \lhd G \ \leftrightharpoons \ Kg = gK\), so there's a \emph{partial commutativity}, if \(g \in G\) and \(k \in K, \ \exists k' \in K : gk = k'g\). 
\end{note}

\begin{definition}
    A \emph{conjugate} of \(x\in G\) is an elemente of the form \(axa^{-1} : a \in G\). 
\end{definition}

\begin{theorem}
    \textbf{2.21.} If \(N \lhd G\), then the cosets form a group, the \emph{quotient group}, denoted by \(\nicefrac{G}{N}\), of order \([G:N]\). \comentario{Set \(Na * Nb := NaNb\) and prove the properties  }
\end{theorem}

\begin{corollary}
    \textbf{2.22.} If \(N \lhd G\), then the \emph{natural map}, \(\pi: G \to \nicefrac{G}{N}\) such that \( a\mapsto Na\), is a surjective homomorphism with \(\ker \pi = N\). \comentario{It's clear}
\end{corollary}

\begin{note}
    The quotient is the generalization of the construction of \(\Z_n\) as \(\nicefrac{\Z}{\langle n \rangle}\) and all normal subgroup is kernel of some homomorphism.  
\end{note}

\begin{definition}
    Given \(a,b \in G\), we denote the \emph{commutator} as \([a,b]:= aba^{-1}b^{-1}\) and the \emph{commutator subgroup} as \(G'\), the generated by all the commutators. 
\end{definition}

\alerta{\begin{example}
    \(X := \{[a,b] : a,b \in G\}\) need not to be a subgroup as \([a,b] * [c,d]\) need not to be a commutator (\textcolor{rojoscuro}{Ex. 43}).
\end{example} 
    }

\begin{theorem}
    \textbf{2.23.} \(G' \lhd G\). Moreover, if \(H\lhd G\), then \(\nicefrac{G}{H}\) is abelian \ \(\leftrightharpoons\) \ \(G' \leq H\). 
\end{theorem}

\demo{If \(f : G \to G \) is homomorphism, then \(f([a,b]) = [f(a), f(b)]\) so \(G' \lhd G\) (\textcolor{rojoscuro}{Ex. 34}); Moreover, \(Hab = Hba \ \leftrightharpoons \ (ab)(ba)^{-1} = a b a^{-1}b^{-1} = [a,b] \in H\).}

\begin{theorem}
    \textbf{2.26. (\emph{Second Isomorphism Theorem})} Let \(T, N \leq G\) with \(N\) normal. Then \(N \cap T\lhd T\) and \(\nicefrac{T}{N\cap T}\cong \nicefrac{NT}{N}\). \comentario{By \(\blacksquare\) \hyperlink{thm224}{2.24. (\emph{FIT})} with \(\ker \pi|_T = N \cap T \lhd T\) and \(\nicefrac{T}{N\cap T} \cong \operatorname{Im}\pi|_T = \nicefrac{NT}{N}\)}
\end{theorem}



\begin{theorem}
    \textbf{2.27. (\emph{Third Isomorphism Theorem})} Let \(K \leq H\) where both \(K, H \lhd G\). Then \(\nicefrac{H}{K}\lhd \nicefrac{G}{K}\) and \((\nicefrac{G}{K}) \ \slash \ (\nicefrac{H}{K}) \cong \nicefrac{G}{H}\). \comentario{Follows by \(\blacksquare\) \hyperlink{thm224}{2.24. (\emph{FIT})}. with \(f: \nicefrac{G}{K} \twoheadrightarrow \nicefrac{G}{H}\  \mid \  Ka \mapsto Ha\) and \(\ker f = \nicefrac{H}{K}\) }
\end{theorem}



\begin{theorem}
    \textbf{2.28. (\emph{Correspondence Theorem})} Let \(K \lhd G\) and \(\pi : G \to \nicefrac{G}{K}\). Then \(S \mapsto \pi(S) = \nicefrac{S}{K}\) is a bijection between \(\{S: K \leq S \leq G\}\) and \(\{H : H \leq \nicefrac{G}{K}\}\). Moreover, if we denonte \(S^* = \nicefrac{S}{K}\), then: 
    \begin{enumerate}[label = \roman*., left = 0.35cm]
        \item \vspace{-0.2cm}\(T \leq S \ \leftrightharpoons \ T^* \leq S^*\), and then \([S:T] = [S^*: T^*]\). 
        \item \(T \lhd S \ \leftrightharpoons \ T^* \lhd S^*\), and then \(\nicefrac{S}{T}= \nicefrac{S^*}{T^*}\). 
    \end{enumerate}
\end{theorem}

\comentario{Doesn't support summaries, see \cite[pag. 38]{stein}}

\begin{definition}
    A group \(\id < G\) is \emph{simple} if \(\nexists K \lhd G : \id < K < G\).  
\end{definition}

\subsection*{Direct Products}

\begin{definition}
    Given \(H\) and \( K \) groups, their \emph{direct product}, denoted by \(H \times K\) is a group with the operation (by coordinates), \   \(*: (h,k) \times (h',k') \mapsto (h h', k k')\). 
\end{definition}

\begin{note}
    \(H \cong H \times \id_K\) and \(K \cong \id_H \times K\). 
\end{note}

\begin{theorem}
    \textbf{2.29.} Let \(H, K \lhd G\). If \(HK = G \) and \(H \cap K = \id\), then \(G \cong H \times K\).  
\end{theorem}

\demo{Let \(f: G \to H \times K  \) such that \( a = hk \mapsto (h,k)\), use commutators to see that \(h'kh'^{-1}k^{-1} \in H \cap K\) and prove the remaining properties. }

\alerta{ \centering All hypothesis in \(\blacksquare\) \hyperlink{thm229}{2.29.} are necessary
\tcblower
\begin{example}
    \(G = S_3\) (non abelian), \(H = \langle (1 \ \ 2 \ \ 3)\rangle \) and \(K = \langle (1 \ \ 2) \rangle\). Check everything except by \(K\ntrianglelefteq G\). We have \(H \times K \cong \Z_3 \times \Z_2\) (abelian). 
\end{example}
}

\begin{theorem}
    \textbf{2.30.} If \(A \lhd H\) and \(B \lhd K\), then \(A \times B \lhd H \times K\) and \(\nicefrac{H\times K}{A \times B} \cong \nicefrac{H}{A}\times \nicefrac{K}{B}\). 
\end{theorem}

\demo{Follows by \(\blacksquare\) \hyperlink{thm224}{2.24. (\emph{FIT})} with \(\varphi : H\times K \twoheadrightarrow \nicefrac{H}{A} \times \nicefrac{K}{B}\) such that \((h,k) \mapsto (Ah, Bk)\) and \(\ker \varphi = A \times B\). 
} 

\begin{corollary}
    \textbf{2.31.} If \(G = H \times K \), then \(\nicefrac{G}{H\times \id}\cong \nicefrac{H}{H} \times \nicefrac{K}{\id} \cong K\).  
\end{corollary}

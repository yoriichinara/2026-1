\section{Los Teoremas de Sylow (\(|G|< \infty\))}

\subsection*{$p$-Grupos}
\begin{definition}
    Siendo \(p\) primo, decimos que \(G\) es $p$\emph{-grupo} si \(\exists a \in G : |a| = p \). 
\end{definition}

\hypertarget{lemma41}{}
\begin{lemma}
    \textbf{4.1.} Si \(G\) es abeliano tal que \(|G|= pm\), entonces \(G\) es $p$-grupo.  
\end{lemma}

\demo{Por inducción en \(m\), sea \(x \in G : |x| = t >1\). Si \(p \mid t\), \(|x^{\nicefrac{t}{p}}| = p\) (\textcolor{rojoscuro}{Ex. 2.11}), en otro caso \(\langle x\rangle \lhd G\) e \(G^*:= \nicefrac{G}{\langle x\rangle}\) es abeliano \( : |G^*| = pm', \ m' < m\), luego \(\exists a^* = \pi(a) \in G^* : |a^*| = p\), tenemos \(|a| = pk\) (\textcolor{rojoscuro}{Ex. 2.14}) e \(|a^{k}|= p\).    }

\begin{theorem}
    \textbf{4.2.} Si \(|G| = pm\), entonces \(\exists a : |a| = p\).  
\end{theorem}

\demo{Si \(\exists x \in G \setminus Z(G): p \) divide \(|C_G(x)|\), el resultado sigue por inducción. En otro caso \(p\) divide \([G : C_G(x)]\) e escogiendo representantes adecuados tenemos, 
\[ \circledast \hspace{1cm}  |G| = |Z(G)| + \sum [G: C_G(x_j)] \ \ \leftarrow \text{\emph{Ecuación de clases} de \(G\)}\]
Luego \( p \) divide \(|Z(G)|\) \text{ (abeliano)} y el resultado sigue del {\scriptsize \(\square\)} \hyperlink{lemma41}{4.1.}.
}

\hypertarget{coro43}{}
\begin{corollary}
    \textbf{4.3.} \(G\) es $p$-grupo \ \(\leftrightharpoons\) \ \(|G| = p^k\). \comentario{Mogollas}
\end{corollary}

\hypertarget{thm44}{}
\begin{theorem}
    \textbf{4.4.} Si \(\id < G\) es $p$-grupo, entonces \(\id < Z(G)\). \comentario{Por el {\scriptsize\(\boxtimes\)} \hyperlink{coro43}{4.3.} todo \(C_G(x_i) \) es $p$-grupo, luego de \(\circledast\) \(|Z(G)| = p^k, \ k>1\)}
\end{theorem}

\begin{corollary}
    \textbf{4.5.} Si \(|G|= p^2\), entonces \(G\) es abeliano. \comentario{Si \(|Z(G)| = p \), entonces \(\nicefrac{G}{Z(G)} \cong \Z_p\) cíclico.\Lightning \ (\textcolor{rojoscuro}{Ex. 3.3}) Por lo tanto, \(|Z(G)| = p^2\) and \(G = Z(G)\) abelian} 
\end{corollary}

\begin{theorem}
    \textbf{4.6.} Si \(|G|= p^k\), entonces (i) Si \(H < G\), entonces \(H < N_G(H)\) e (ii) Todo \(H \leq G\) maximal es normal e \([G : H ] = p\). \comentario{Véase \cite[pag. 75]{stein} e (\textcolor{rojoscuro}{Ex. 2.58})}
\end{theorem}

%\demo{(i) If \(H\) is not normal and \(X:= \{H^g : g \in G\}\), then \(|X| = [G: N_G(H)] > 1 \)}

\begin{theorem}
    \textbf{4.8.} Si \(|G|= p^k\) e \(r_s := \#\{H \leq G : |H| = p^s\}\), entonces \(r_s \equiv 1 \ (\operatorname{mod }p)\). \comentario{No admite resumen, véase \cite[pag. 76]{stein}}
\end{theorem}

\subsection*{Los Teoremas de Sylow}

\begin{definition}
    \(P\leq G\) es \emph{subgrupo $p$-Sylow} si es $p$-subgroup maximal de \(G\).  
\end{definition}

\hypertarget{lemma411}{}
\begin{lemma}
    \textbf{4.11.} Si \(P \leq G\) es $p$-Sylow, entonces (i) \((|\nicefrac{N_G(P)}{P}|, p) = 1 \) e (ii) Si \(|a| = p^k\) e \(P^a = P\), entonces \(a \in P\). \comentario{Véase \cite[pag. 78]{stein}}  
\end{lemma}

\begin{theorem}
    \textbf{4.12. (\emph{Sylow})} Sea \(P\leq G\) $p$-Sylow e \(n_p := \#\{H \leq G : H \text{ es }p\text{-Sylow}\}\), entonces
    \vspace{-0.2cm}
    \begin{enumerate}[label = \roman*., left = 0.2cm]
        \item \(\forall H\leq G\) $p$-Sylow, \(\exists g \in G : H = P^g\). \hspace{1cm} ii. \( n_p \equiv 1 \ (\operatorname{mod}\ p)\) and \(n_p \mid |G|\).
    \end{enumerate}
\end{theorem} 

\vspace{-0.2cm}
\comentario{No admite resumen, véase \cite[pag. 79]{stein}}

\begin{corollary}
    \textbf{4.13.} \(\exists ! P\leq G\) \(p\)-Sylow \(\ \leftrightharpoons \ \) \(P \lhd G\).  \comentario{Mogollas}
\end{corollary}

\hypertarget{thm414}{}
\begin{theorem}
    \textbf{4.14.} Si \(|G| = p^em\), donde \((p, m) = 1\), entonces \(\forall P \leq G\) \(p\)-Sylow, \(|P|= p^e\).  
\end{theorem}

\demo{\(N:= N_G(P)\), \([G:P] = [G:N][N:P]\), donde \([G:N] = n_p \equiv 1 \ (\operatorname{mod} \ p)\) mientras \([N:P] = |\nicefrac{N}{P}|\) es coprimo con \(p\), {\scriptsize \(\square\)} \hyperlink{lemma411}{4.11.}. By \(\blacksquare\) \hyperlink{thm211}{2.11. (\emph{Lagrange})} \(|P| = p^k, \ k \leq e\) y \([G:P] = \nicefrac{|G|}{|P|} = p^{e-k}m\) coprimo con \(p \ \leftrightharpoons \ k=e\).}

\hypertarget{coro415}{}
\begin{corollary}
    \textbf{4.15.} Si \(p^k \) divide \(|G|\), entonces \(\exists H\leq G : |H| = p^k\). \comentario{Sea \(P \leq G\) \(p\)-Sylow, entonces \(p^k \) divide \(|P|\) y el resultado sigue (\textcolor{rojoscuro}{Ex. 4.2})}
\end{corollary}

\subsection*{Grupos de Pequeño Orden}

\begin{theorem}
    \textbf{4.19.} Todo grupo de orden \(2p\) es cíclico o dihedral. 
\end{theorem}

\demo{Caso \(p = 2\) (\textcolor{rojoscuro}{Ex. 2.12}). Si \(p>2\), por el {\scriptsize\(\boxtimes\)} \hyperlink{coro415}{4.15.} e \(\blacksquare\) \hyperlink{thm211}{2.11. (\emph{Lagrange})} \(\exists s,t \in G : |s|=p \) e \(|t|= 2\), sigue que \(\langle s \rangle \lhd G\) e \(tst = s^i\). Más aún, \(s = t(t s t)t = s^{i^2} \ \leftrightharpoons \ i^2 \equiv 1 \ (\operatorname{mod }p) \ \leftrightharpoons \ i = \pm 1\). Luego, \(G\cong \Z_2 \times \Z_p \cong \Z_{2p}\) o \(G \cong D_{2p}\).   }

\begin{theorem}
    \textbf{4.20.} Si \(|G|= pq\), con \(p>q\) primos, entonces \(\exists N \lhd G : |N| = p\). Más aún, si \(q \nmid p-1\), entonces \(G\) es cíclico. 
\end{theorem}

\demo{\(n_p \equiv 1 \ (\operatorname{mod} p)\) e \(n_p \mid q \ \leftrightharpoons \ n_p =1\). Más aún, si \(q \nmid p-1\), entonces \(n_q \equiv 1 \ (\operatorname{mod } q)\) e \(n_q \neq p \ \leftrightharpoons \ n_q =1\). Sigue del \(\blacksquare\) \hyperlink{thm229}{2.29.} que \(G = \Z_p \times \Z_q \cong \Z_{pq}\).  }

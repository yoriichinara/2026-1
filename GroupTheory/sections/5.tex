\section{Series Normales (\(|G|< \infty\))}

\begin{definition}
    Una \emph{serie normal} de \(G\) es una sucesión \(G = G_0 \geq G_1 \geq \cdots \geq G_n = \id\), donde \(G_{i+1}\lhd G_i, \ \forall i \leq n-1\). Los grupos \(\nicefrac{G_{i+1}}{G_i}\) son llamados \emph{factores} de la serie, y el \emph{comprimento} es la cantidad de inclusiones estrictas.
\end{definition}

\begin{note}
    Ser normal no es una propiedad transitiva (\textcolor{rojoscuro}{Ex. 2.45}), por ello podemos asegurar factores apenas en elementos contiguos de la serie. 
\end{note}

\subsection*{Teorema de Jordan-Hölder}
\begin{definition}
    Un grupo \(G\) es \emph{soluble} se \(\exists (G_j)_{j\leq n}\) serie normal tal que todos sus factores son grupos cíclicos de orden primo. Tambien, 
    \vspace{-0.3cm}
    \begin{itemize}[label = \(\star\), left = 1cm]
        \item \((H_k)_{k\leq m}\) (serie normal en \(G\)) es un \emph{refinamiento} de \((G_j)\) si \(G_0, G_1, \ldots, G_n\) es subsucesión de \(H_0, H_1, \ldots, H_m\).
        \item \((G_j)_{j\leq n}\) es \emph{serie de composición} si cada \(G_{i+1}\lhd G_i\) es maximal \(G_i\). En este caso tenemos \emph{factores de composición}.     
    \end{itemize}
\end{definition}

\vspace{-0.3cm}
\begin{definition}
    Dos series normales \((G_j), (H_k)\) en \(G\) son \emph{equivalentes} si existe una correspondencia biyectiva entre factores isomorfos. 
\end{definition}

\begin{lemma}
    \textbf{5.10. (\emph{Zassenhaus, 1934})} Sean \(A \lhd A^* \) e \(B \lhd B^*\) subgrupos de \(G\). Entonces, \(A (A^* \cap B) \lhd A(A^* \cap B^*); \ \  B (B^* \cap A) \lhd B(B^* \cap A^*)\) e
    \[\exists \varphi : \displaystyle \frac{A(A^*\cap B^*)}{A(A^*\cap B)} \cong \frac{B(B^*\cap A^*)}{B(B^*\cap A)} .\]
\end{lemma}

\vspace{-0.3cm}
\comentario{No admite resumen, véase \cite[pag. 100]{stein}}

\hypertarget{thm511}{}
\begin{theorem}
    \textbf{5.11. (\emph{Refinamiento de Schreier, 1928})} Cualesquier dos series normales en \(G\) admiten refinamientos que son equivalentes. \comentario{Véase \cite[pag. 100]{stein}} 
\end{theorem}

\begin{theorem}
    \textbf{5.12. (\emph{Jordan-Hölder})} Series de composición en \(G\) son equivalentes. \comentario{Sigue directo del \(\blacksquare\) \hyperlink{thm511}{5.11. (\emph{Schreier})} pues solo tienen refinamientos triviales} 
\end{theorem}

\begin{corollary}
    \textbf{5.13. (\emph{Teorema Fundamental de la Aritmética})} Factores primos e multiplicidades en la descomposición de cualquier \(n\in \Z_{\geq 2}\), depende solo de \(n\).    
\end{corollary}

\demo{
    Sean \(n = p_1 p_2 \cdots p_m\) (no necesariamente distintos) e \(\Z_n = \langle x \rangle \geq \langle x^{p_1}\rangle \geq \langle x^{p_1p_2}  \rangle \geq \cdots \geq\langle x^{p_1p_2 \cdots  p_{m-1}}\rangle \geq \id\), los factores tienen order \(p_1, p_2, \ldots, p_m\), luego es serie de composición y solo depende de \(n\), \(\blacksquare\) \hyperlink{thm512}{5.12. (\emph{Jordan-Hölder})}. 
}

\begin{theorem}
    \textbf{5.14.} Si \(n\geq 5\), entonces \(S_n\) no es soluble. \comentario{\(A_n\) es simple, luego \(S_n \geq A_n \geq \id \) es serie de composición de factores \(\Z_2\) e \(A_n\) (no abeliano)}
\end{theorem}

\subsection*{Grupos Solubles}    

\begin{definition}
    Una \emph{serie soluble} de un grupo \(G\) es una serie normal cuyos factores son abelianos, \(G\) va a ser \emph{soluble} si existíera tal serie. \footnote{La definición es compatible, todo \(\Z_m\) se puede refinar a cíclicos de orden primo.  }
\end{definition}

\hypertarget{thm515}{}
\begin{theorem}
    \textbf{5.15.} Subgrupo de grupo soluble es de nuevo soluble.
\end{theorem}

\demo{Sea \((G_j)_{j\leq n}\) serie soluble de \(G\), entonces la serie \((H\cap G_j)\) es soluble en \(H\). Sigue del \(\blacksquare\) \hyperlink{thm226}{2.26.}, pues \(H \cap G_{i+1} = (H \cap G_i) \cap G_{i+1} \lhd H\cap G_i\) e 
\[\frac{H\cap G_i}{H\cap G_{i+1}} \cong G_{i+1} \frac{H\cap G_i}{G_{i+1}} \leq \frac{G_i}{G_{i+1}} \ \text{(abeliano)}.\]
\vspace{-0.3cm}
}

\hypertarget{thm516}{}
\begin{theorem}
    \textbf{5.16.} Cociente de grupo soluble es de nuevo soluble. \comentario{Véase \cite[pag. 102]{stein}}
\end{theorem}

\hypertarget{thm517}{}
\begin{theorem}
    \textbf{5.17.} Si \(H \lhd G\) y ambos \(H \) e \(\nicefrac{G}{H}\) son solubles, entonces \(G\) es soluble.
\end{theorem}

\demo{Sea \((K_j^*)_{j\leq m}\) serie soluble en \(H\). Por el \(\blacksquare\) \hyperlink{thm228}{2.28. (\emph{Correspondencia})} \(\exists (K_j) : G \geq K_1 \geq \cdots \geq K_n = H \), con \(K_{i+1} \lhd K_i\) e \(\nicefrac{K_i}{K_{i+1}} \cong \nicefrac{K_i^*}{K_{i+1}^*}\) (abeliano). Por lo tanto, la serie \(G,K_1, \ldots, K_n=H=K_0^*, K_1^*, \ldots, K_n^* = \id \) es soluble en \(G\). } 

\begin{corollary}
    \textbf{5.18.} Si \(H\) e \(K\) son grupos solubles, entonces \(H\times K\) es soluble. \comentario{ Sigue directo del \(\blacksquare\) \hyperlink{thm517}{5.17.} pues \(H\cong H\times \id  \lhd H\times K\) e \(\nicefrac{H\times K}{H\times \id} \cong \id \times K \cong K \)}
\end{corollary}

\begin{theorem}
    \textbf{5.19.} Todo \(p\)-grupo es soluble. \comentario{Por el \(\blacksquare\) \hyperlink{thm44}{4.4.}, \(\id < Z(G)\) abeliano (soluble), sigue inducción en \(|G|\) pues \(|\nicefrac{G}{Z(G)}|< |G|\), luego vale \(\blacksquare\) \hyperlink{thm517}{5.17.}}
\end{theorem}

\begin{definition}
    Los \emph{subgrupos de conmutadores de mayor orden} son definidos inductivamente por \(G^{(i+1)} = G^{(i)}{'}\). La serie \(G =: G^{(0)}\geq G^{(1)} \geq \cdots\) es llamada \emph{serie derivada} de \(G\). 
\end{definition}

\begin{definition}
    Un \emph{automorfismo} es un \(\varphi: G \cong G\). Tambien, \(H \leq G\) es \emph{característico}, \(H \operatorname{char} G\) si \(\varphi(H) = H\) para cada automorfismo \(\varphi \) de \(G\). 
\end{definition}

\begin{note}
    Dado \(a \in G\) fijo, la conjugación \(x \mapsto axa^{-1} \) es un automorfismo luego todo característico es subgrupo normal más el contrário no es cierto (\textcolor{rojoscuro}{Ex. 5.28}).
\end{note}

\hypertarget{lemma520}{}
\begin{lemma}
    \textbf{5.20.} (i) Si \(H \operatorname{char} K \operatorname{char} G\), entonces \(H \operatorname{char} G\); (ii) Se \(H \operatorname{char} K \lhd G\), entonces \(H \lhd G\). \comentario{(i) \(\varphi(H) = \varphi|_{_K}(H)= H\) e (ii) \(\varphi: x \mapsto axa^{-1}\) e \(\varphi|_{_K}(H) \leq H \)} 
\end{lemma}

\begin{theorem}
    \textbf{5.21.} En todo \(G\), los subgrupos \(G^{(i)} \operatorname{char} G\), luego en \(G^{(i)} \lhd G, \forall i\geq 0\). 
\end{theorem}

\demo{Por inducción en \(i\), dado \(\varphi\) automorfismo \(\varphi([a,b]) = [\varphi(a), \varphi(b)]\), luego \(\varphi(G') \leq G'\) e \(G^{i+1}\operatorname{char}G^{(i)}\operatorname{char} G\), sigue del {\scriptsize \(\square\)} \hyperlink{lemma520}{5.20. (ii)} que \(G^{(i+1)} \operatorname{char} G\).}

\hypertarget{lemma522}{}
\begin{lemma}
    \textbf{5.22.} Si \((G_j)_{j\leq n}\) es una serie soluble en \(G\), entonces \(G_i \geq G^{(i)}, \ \forall i\leq n\). En particular, \(\id \geq G^{(n)}\geq \id\). 
\end{lemma}

\demo{Por inducción en \(i\), como \(\nicefrac{G_i}{G_{i+1}}\) es abeliano, sigue del \(\blacksquare\) \hyperlink{thm223}{2.23.} e hipótesis inductiva que \(G_{i+1}\geq G_{i}'\geq G^{(i)}{'} = G^{(i+1)}\). }

\hypertarget{thm223}{}
\begin{theorem}
    \textbf{5.23.} Un grupo \(G\) es soluble \(\ \leftrightharpoons \ \) \(G^{(n)} = \id\) para algún \(n \in \N\). 
\end{theorem}

\demo{(\(\Rightarrow\)) {\scriptsize \(\square\)} \hyperlink{lemma522}{5.22.}; (\(\Leftarrow\)) Ya vimos que la serie es normal, ahora converge e los factores son abelianos pues \(G^{(i+1)}\lhd G^{(i)}\) e \(G^{(i)}{'} \leq G^{(i)}\), \(\blacksquare\) \hyperlink{thm223}{2.23.}. }

\begin{note}
    El {\scriptsize\(\square\)} \hyperlink{lemma522}{5.22.} muestra que la serie derivada converge más rápido que cualquiera e el \(\blacksquare\) \hyperlink{thm523}{5.23.} muestra que esta serie es normal \(\ \leftrightharpoons \ \) \(G\) es soluble. Es posible reversionar e demostrar los \(\blacksquare\) \hyperlink{thm515}{5.15.}, \(\blacksquare\) \hyperlink{thm516}{5.16.} e \(\blacksquare\) \hyperlink{thm517}{5.17.} con estas consideraciones. 
\end{note}

\subsection*{Series Centrales y Grupos Nilpotentes}

En esta parte introducimos y recuperamos algunas definiciones de secciones anteriores, valiendónos principalmente de la siguiente: 
\begin{definition}
    Dados \(H,K \leq G\) definimos \(\displaystyle [H,K]:= \langle[h,k] : h \in H, k \in K \rangle\). Tambien, 
    \vspace{-0.3cm}
\begin{itemize}[label = \(\star\), left = 1cm]
    \item \(K < N_G(H) \leftarrow \ K\) \emph{normaliza} \(H\)\textcolor{gray}{\( \  \leftrightharpoons \ [H,K ] \leq H\)}
    \item \(C_G(H)= \{x \in G : [x,h] = \id, \ \forall h \in H\leq G\}\) 
    \item \(K < C_G(H) \leftarrow \ K \) \emph{centraliza} \(H\) \textcolor{gray}{\( \  \leftrightharpoons \ [H,K ] = \id\)}
    \item \([x,y] \in K \lhd G \leftarrow \ x  \) e \(y\) \emph{conmutan} \(\operatorname{mod}  K \) (como clases en \(\nicefrac{G}{K}\))
\end{itemize}
\end{definition}

\begin{note}
    Es claro que \([H,K] = [K,H]\), pues \([h,k]^{-1} = (hkh^{-1}k^{-1})^{-1} = [k,h] \). Tambien, en está nueva notación, \(G^{(i+1)} = [G^{(i)},G^{(i)}]\). 
\end{note}

\hypertarget{lemma530}{}
\begin{lemma}
    \textbf{5.30.} (i) Si \(K \lhd G\) e \(K \leq H \leq G\), entonces \([H,G] \leq K \ \leftrightharpoons \ \nicefrac{H}{K} \leq Z(\nicefrac{G}{K})\) e (ii) Si  \(f : G \to L\) es homomorfismo, entonces \(f([H,K]) = [f(H), f(K)]\).   
\end{lemma}

\demo{(i) \(hKgK = gKhK \ \leftrightharpoons \ [h,g]K = K \ \leftrightharpoons \ [h,g] \in K \); (ii) Ambos lados son generados por los elementos \(f([h,k]) = [f(h), f(k)]\). }

\begin{definition}
    Considere los subgrupos característicos de \(G\), definidos inductivamente por:
    %\(\star \ \  \leftarrow \) \ \emph{Subgrupos característicos} de \(G\)
    \vspace{-0.33cm}
    \begin{itemize}[label = \(\star\), left=1cm]
        \item \(\gamma_1 := G; \ \  \gamma_{i+1} := [\gamma_i, G]\)
        \item \(\zeta^0 := \id; \ \  \zeta^{i+1} := \nicefrac{\zeta^{i+1}}{\zeta^i} = Z(\nicefrac{G}{\zeta^i})\leftarrow \ \) \emph{Centros de orden superior}
        \item\( G = \gamma_1 \geq \gamma_2 \geq \cdots  \leftarrow \ \) \emph{Serie central inferior} (\emph{descendente})
        \item \(\id = \zeta^0 \leq \zeta^1 \leq \cdots  \leftarrow \ \) \emph{Serie central superior} (\emph{ascendente})
    \end{itemize}
\end{definition}

\begin{note}
    Es claro que \(\nicefrac{\gamma_i(G)}{\gamma_{i+1}(G)}\leq Z(\nicefrac{G}{\gamma_{i+1}(G)})\), {\scriptsize \(\square\)} \hyperlink{lemma530}{5.30. (i)} e tambien \(\zeta^1 = Z(G)\). Cabe notar que la serie inferior no tiene porque ser normal (puede no conveger). 
\end{note}

\hypertarget{thm531}{}
\begin{theorem}
    \textbf{5.31.} En un grupo \(G\), \(\exists c \in \Z_{\geq 0} : \zeta^c = G \ \leftrightharpoons \ \gamma_{c+1}= \id \). Más aún, en este caso \(\gamma_{i+1} \leq \zeta^{c-i}, \ \forall i \leq c\). \comentario{No admite resumen, véase \cite[pag. 113]{stein}}
\end{theorem}

\begin{theorem}
    \textbf{5.32. (\emph{Schur})} Si \(|\nicefrac{G}{Z(G)}|< \infty\), entonces \(|G'| < \infty\). \comentario{Véase \cite[pag. 114]{stein}}  
\end{theorem}

\begin{definition}
    Un grupo \(G\) es \emph{nilpotente} si \(\exists c \in \Z_{\geq 0} \) (\emph{clase de nilpotencia}) \(: \gamma_{c+1} = \id\).
\end{definition}

\begin{note}
    Para grupos nilpotentes las series superior e inferior son normales de misma longitud, si \(c= 2\), \(\gamma_2 = G' \leq Z(G) = \zeta^1\). Tambien, \(c = 1 \ \leftrightharpoons \ G\) es abeliano.
\end{note}

\hypertarget{thm533}{}
\begin{theorem}
    \textbf{5.33.} Todo \(p\)-grupo finito es nilpotente. \comentario{Si \(\zeta^i < G\), entonces \(\id < Z(\nicefrac{G}{\zeta^i})\), \(\blacksquare\) \hyperlink{thm44}{4.4.}, luego \(\zeta^i < \zeta^{i+1}\), como \(|G| < \infty\), \(\exists c : \zeta^c = G\) }
\end{theorem}

\begin{note}
    El caso infinito no es necesariamente cierto, sin embargo existen $p$-grupos infinitos nilpotentes, McLain (1954). %ió uno donde \(Z(G) = \id \) e  \(G' = G\) son sus únicos subgrupos característicos.
\end{note}

\begin{theorem}
    \textbf{5.34.} (i) Todo nilpotente es soluble; (ii) Si \(\id < G\) es nilpotente, entonces \(\id < Z(G)\) e (iii) \(S_3\) es un grupo soluble que no es nilpotente. 
\end{theorem}

\demo{
    (i) \(G^{(i)} \leq \gamma_i\) (inducción), luego \(G^{(c+1)} \leq \gamma_{c+1} = \id\); (ii) \(\id <  \gamma_c =  \gamma_{c-1+1}  \leq \zeta^{c-(c-1)} = \zeta^1 = Z(G)\), \(\blacksquare\) \hyperlink{thm531}{5.31.}; (iii) \(S_3\) es soluble, mas \(Z(S_3) = \id \). 
}

\begin{theorem}
    \textbf{5.35.} Si \(H\leq G\) nilpotente es nilpotente e \(c_{_H} \leq c_{_G}\). \comentario{\(\gamma_i (H) \leq \gamma_i(G)\) (inducción), luego \(\gamma_{c+1}(H) \leq\gamma_{c+1}(G) = \id\)}
\end{theorem}

\begin{theorem}
    \textbf{5.36.} Si \(N\lhd G\) nilpotente, entonces \(\nicefrac{G}{N}\) es nilpotente e \(c_{_N} \leq c_{_G}\). \comentario{\(f : G \twoheadrightarrow L\) homomorfismo implica \(\gamma_i(L) \leq f(\gamma_i(G))\), {\scriptsize \(\square\)} \hyperlink{lemma530}{5.30.}, luego \(\gamma_{c+1}(L) \leq \gamma_{c+1}(G) = \id\) }
\end{theorem}

\begin{note}
    El análogo del \(\blacksquare\) \hyperlink{thm517}{5.17.} no es cierto en general, por ejemplo, en \(G = S_3\) tenemos \(A_3 \cong \Z_3\) e \(\nicefrac{S_3}{A_3} \cong \Z_2\) (abelianos, nilpotentes), precisa más hipótesis (\textcolor{rojoscuro}{Ex. 5.38}).    
\end{note}

\hypertarget{thm537}{}
\begin{theorem}
    \textbf{5.37.} Si \(H\) e \(K\) son nilpotentes, entonces \(H\times K\) es nilpotente. \comentario{\(\gamma_i(H\times K) \leq \gamma_i(H) \times \gamma_i(K)\) (inducción), luego \(\gamma_{M+1} (H\times K) \leq \id_{_H}\times \id_{_K}, \ M:= \max\{c_{_H}, c_{_K}\}\)}
\end{theorem}

\hypertarget{thm538}{}
\begin{theorem}
    \textbf{5.38.} Si \(G\) es nilpotente, entonces satisface la \emph{condición normalizadora}: si \(H< G\), entonces \(H < N_G(H)\). 
\end{theorem}

\demo{\(\exists i : \gamma_{i+1}\leq  H < \gamma_i\), luego \([\gamma_i, H] \leq [\gamma_i, G] = \gamma_{i+1} \leq H\) e \(H < \gamma_i \leq N_G(H)\). La vuelta tambien vale (\textcolor{rojoscuro}{Ex. 5.37}).} 

\hypertarget{thm539}{}
\begin{theorem}
    \textbf{5.39.} \(G : |G| < \infty\) es nilpotente \( \ \leftrightharpoons \  G = \prod P_i: P_i \leq G\) es $p_i$-Sylow. 
\end{theorem}

\demo{(\(\Rightarrow\)) \(N_G(P) =: H = N_G(H)\) (\textcolor{rojoscuro}{Ex. 4.11}). Si \(N_G(P)<G\), entonces \(N_G(P) < N_G(H)\), \(\blacksquare\) \hyperlink{thm538}{5.38.}.\Lightning \ Luego, \(N_G(P) =  G\ \leftrightharpoons \ P \lhd G\) e el enunciado sigue (\textcolor{rojoscuro}{Ex. 4.12}); (\(\Leftarrow \)) Sigue de los \(\blacksquare\) \hyperlink{thm533}{5.33.} e \(\blacksquare\) \hyperlink{thm537}{5.37.}.}

\begin{theorem}
    \textbf{5.40.} En \(G\) nilpotente todo \(H\leq G\) maximal es normal de índice primo. \comentario{\(H< N_G(H) = G \ \leftrightharpoons \ H\lhd G\), \(\blacksquare\) \hyperlink{thm538}{5.38.}, \(|\nicefrac{G}{H}| = p\) sigue del (\textcolor{rojoscuro}{Ex. 2.58})}
\end{theorem}

\begin{theorem}
    \textbf{5.41.} Sea \(G\) grupo nilpotente, entonces (i) Si \(\id < H \lhd G\), entonces \(\id < H \cap Z(G) \)  e (ii) Si \(A \lhd G\) es maximal abeliano, entonces \(A = C_G(A)\). \comentario{Véase \cite[pag. 117]{stein}}
\end{theorem}

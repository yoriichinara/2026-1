\subsection*{Cauchy's integral formula}

\hypertarget{thm241}{}
\parbox[a]{0.7\linewidth}{
\begin{theorem}
    \textbf{4.1.} If \(f\) is holomorphic in \(\Omega\ab\subset \C\) and \(D : \overline{D}\subset \Omega\), then 
    \[f(z) = \frac{1}{2\pi i } \int_{\partial D} \underset{\textcolor{gray}{\overset{\uparrow }{F(\zeta)}}}{\frac{f(\zeta)}{\zeta - z}} \ d\zeta, \  \ z \in D. \]
\end{theorem}
}

\demo{
        \[0 =  \lim_{\epsilon \to 0} \int_{\Gamma_{\delta, \epsilon}} F(\zeta) \ d\zeta = \int_{\partial D} F(\zeta ) \ d\zeta + I_\epsilon  + \int_{\partial D_\epsilon} \frac{f(z)}{\zeta - z} \ d\zeta. \]
        \hrule 
        \vspace{0.3cm}
        \textcolor{gray}{\[I_\epsilon = \int_{\partial D_\epsilon} \frac{f(\zeta) - f(z)}{\zeta - z} \ d\zeta \to 0 \text{ \ and \ }\int_{\partial D_\epsilon} \frac{f(z)}{\zeta - z} \ d\zeta = - f(z) 2\pi i.\]} 
        \vspace{-0.2cm}
}

\hypertarget{coro242}{}
\begin{corollary}
    \textbf{4.2. (\emph{Cauchy integral formula})} If \( f\) is holomorphic in \(\Omega\ab \subset \C\), then \(f\) has infinitely many complex derivatives in \(\Omega\). Moreover, if \(\overline{D}\subset \Omega \), then 
    \[f^{(n)}(z) = \frac{n!}{2\pi i } \int_{\partial D} \frac{f(\zeta)}{\zeta - z} \ d\zeta, \ \ z \in D.\]
\end{corollary}

\hypertarget{coro243}{}
\begin{corollary}
    \textbf{4.3. (\emph{Cauchy inequalities})}  If \( f\) is holomorphic in \(\Omega\ab \subset \C\) and \(B[z_0;R] \subset \Omega \), then 
    \[|f^{(n)}(z_0)|\leq \frac{n! \|f\|_{\partial D}}{R^n}, \text{ where } \|f\|_{\partial D}:= \sup_{\partial D} |f(z)|\]
\end{corollary}

\demo{  
    \[|f^{(n)}(z_0) | = \frac{n!}{2\pi} \left| \int_{\partial D} \frac{f(\zeta )}{\zeta - z} \ d \zeta \right| = \frac{n!}{2\pi} \left|\int_0^{2\pi} (\cdots) \ d\theta \right|\leq \frac{n!}{2\pi}\frac{\|f\|_{\partial D}}{R^n} 2\pi. \]
    \vspace{-0.2cm}
}

\begin{theorem}
    \textbf{4.4.}  If \( f\) is holomorphic in \(\Omega\ab \subset \C\) and \(D = B[z_0;R] \subset \Omega \), then \(f\) has a convergent Taylor power series expansion at \(z_0\): 
    \[f(z) = \sum^\infty_{n=0} a_n (z-z_0)^n, \text{ where } a_n= \frac{f^{(n)}(z_0)}{n!}, \ z\in D.\]
\end{theorem}

\demo{
%    \vspace{-0.2cm}
    \[\frac{1}{\zeta -z} = \frac{1}{\zeta -z_0}\frac{1}{1 -\left(\frac{z-z_0}{\zeta - z_0}\right)} = \frac{1}{\zeta - z_0}\sum_{n=0}^\infty \left(\frac{z-z_0}{\zeta -z_0}\right)^n. \]
    The geometic series converges because \(\nicefrac{\left|z-z_0\right|}{\left|\zeta- z_0\right|} < r <1\) (uniformity). Finally, use \(\blacksquare\) \hyperlink{thm242}{4.2. (\emph{Cauchy integral formula})}  to conclude.
}
